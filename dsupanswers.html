class Appliance(ABC):
    def __init__(self, name):
        self.name = name
        
    @abstractmethod
    def turn_on(self):
        pass

    def turn_on(self):
        print(f"Turning on {self.name}")

    def turn_on(self):
        print(f"Turning on {self.name}")

    count = 0
    for i in word:
        if i == letter:
            count += 1
    return count

from abc import ABC,abstractmethod
class Shape:
    @abstractmethod
    def area(self):
        pass
    @abstractmethod
    def perimeter(self):
        pass
class Triangle(Shape):
    def __init__(self, base, height):
        self.base = base
        self.height = height
    def area(self):
        return 0.5 * self.base * self.height
    def perimeter(self):
        return self.base + self.base + self.height
class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius
    def area(self):
        return 3.14 * self.radius ** 2
    def perimeter(self):
        return 2 * 3.14 * self.radius

triangle =Triangle(5,10)
print(f"Triangle area: {triangle.area()}")
print(f"Triangle perimeter: {triangle.perimeter()}")

circle = Circle(7)
print(f"Circle area: {circle.area()}")
print(f"Circle perimeter: {circle.perimeter()}")
